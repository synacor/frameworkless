Source:

Returns:

The corresponding key's value if it exists, otherwise fallback or undefined.

Examples

Basic Usage:

		var res = {
			json : {
				success : true,
				message : 'Thing completed.'
			}
		};

		var msg = util.delve(res, 'json.message');
		console.log(msg);		// 'Thing completed.'

DOM Traversal:

		document.body.innerHTML = '<span>hello</span>';
		var text = util.delve(window, 'document.body.childNodes.0.textContent');
		console.log(text);		// "hello"

<static> extend(base, props*)

Copy own-properties from props onto base.

Parameters:
Name Type Description
base Object

An object onto which properties should be copied
props* Object

The rest of the arguments will have their properties copied onto base
Source:
  • util.js, line 413
Returns:

base

Examples

Basic Usage:

		var obj1 = {
				test : 'foo',
				obj1prop : 'obj1'
			},
			obj2 = {
				test : 'bar',
				obj2prop : 'obj2'
			};

		// copy properties from obj2 onto obj1:
		util.extend(obj1, obj2);

		console.log(obj1.test);
		//	{
		//		test : "bar",
		//		obj1prop : "obj1",
		//		obj2prop : "obj2"
		//	}

Shallow Clone:

		var orig = {
			foo : 'bar'
		};

		var clone = util.extend({}, orig);

		console.log(clone === orig);   // false
		console.log(clone.foo === orig.foo);   // true

Tip: Cheater constructor options

		function MyClass(options) {
			// copy options onto the instance:
			util.extend(this, options || {});
		}

		new MyClass({
			unique : true
		});

Type

Example
function Animal() {
			this.sound = "";
		}
		Animal.prototype.type = 'Unknown';

		function Cat() {
			// call parent constructor:
			Animal.call(this);
			this.sound = "mew";
		}
		Cat.prototype.type = 'Feline';

		// Make Cat inherit from Animal:
		util.inherits(Cat, Animal);

		var cat = new Cat();
		console.log(cat.sound);		// "mew"
		console.log(cat.type);		// "Feline"
