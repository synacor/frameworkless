Module: events

events

Provides a Class and mixin to enable event firing and listening. If called directly as a function, events() is an alias of events.mixin.

Source:
  • events.js, line 1
See:
  • http://nodejs.org/api/events.html

Example

var events = require('events');

// create a default evented object:
var hub = events();

// or, mix EventEmitter into any existing object:
var hub = { VERSION:1.0 };
events(hub);

// Now you can fire and hook events on your object:
hub.on('hubba', function() {
	console.log('bubba');
});

hub.emit('hubba');		// "bubba"
