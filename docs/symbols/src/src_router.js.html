 40 	function Router() {
 41 		if (!(this instanceof Router)) return new Router();

 44 		this.routes = [];

 52 	/**	Route to a given URL, or add a routing handler for the given URL pattern.
 53 	 *	@param {String} url		A URL to route to
 54 	 *	@param {Function} [handler]		If given, <code>url</code> is used as a URL pattern that maps to this handler function.
 55 	 */
 56 	Router.prototype.route = function(url, handler) {
 57 		if (typeof handler==='function') {
 58 			this.routes.push({
 59 				url : url,
 60 				handler : handler
 61 			});
 62 			this.routes.sort(sort);
 63 			return this;
 64 		}
 65 		history[handler===true ? 'replaceState' : 'pushState'](0, 0, url);
 66 		return this.currentUrl===url || route(this, url);
 67 	};

 75 	// Perform routing for the given router+url combo.
 76 	function route(router, url) {
 77 		var route, matches, i,
 78 			old = router.currentRoute;
 79 		for (i=router.routes.length; i--; ) {
 80 			route = router.routes[i];
 81 			matches = exec(url, route.url);
 82 			if (matches && route.handler(matches)!==false) {
 83 				if (old && typeof old.unload==='function') {
 84 					old.unload();
 85 				}
 86 				router.currentUrl = url;
 87 				router.currentRoute = route;
 88 				if (typeof router.onroute==='function') {
 89 					router.onroute(url);
 90 				}
 91 				return true;
 92 			}
 93 		}
 94 		return false;
 95 	}

 97 	// Check if the given URL matches a route's URL pattern.
 98 	// @returns key-value matches for a match, or false for a mismatch
 99 	function exec(url, route) {
100 		var matches = {};
101 		url = segmentize(url);
102 		route = segmentize(route);
103 		for (var i=0; i<Math.max(url.length, route.length); i++) {
104 			if (route[i].charAt(0)===':') {
105 				matches[route[i].substring(1)] = url[i];
106 			}
107 			else {
108 				if (route[i]!==url[i]) {
109 					return false;
110 				}
111 			}
112 		}
113 		return matches;
114 	}

117 	function segmentize(url) {
118 		return url.replace(/(^\/+|\/+$)/g, '').split('/');
119 	}
